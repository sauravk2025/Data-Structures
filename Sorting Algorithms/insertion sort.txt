#include<stdio.h>

void main()
{
    int i,j,k,a[100],n,temp=0;
    printf("Enter the number of elements:");
    scanf("%d",&n);
    printf("Enter the elements:");
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    printf("The elements are:");
     for(i=0;i<n;i++)
    {
        printf("%d\t",a[i]);
    }
    for(k=1;k<n;k++)
    {
        temp=a[k];
        j=k-1;
        while(temp<=a[j]&&j>=0)
        {
            a[j+1]=a[j];
            j--;
        }
        a[j+1]=temp;
        printf("\nAfter sorting %d:",k);
        for(i=0;i<n;i++)
        {
            printf("%d\t",a[i]);
        }
    }
    
}
